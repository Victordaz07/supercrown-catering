rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: get user role from request
    function isSales() {
      return request.auth != null && request.auth.token.role == 'sales';
    }
    function isDriver() {
      return request.auth != null && request.auth.token.role == 'driver';
    }
    function isClient() {
      return request.auth != null && request.auth.token.role == 'client';
    }
    function isAuthenticated() {
      return request.auth != null;
    }

    // quotes: sales read/write all; clients create + read own (by email)
    match /quotes/{quoteId} {
      allow read: if isSales() || (isAuthenticated() && resource.data.clientEmail == request.auth.token.email);
      allow create: if true;  // Public quote submission (no auth required for clients)
      allow update, delete: if isSales();

      match /items/{itemId} {
        allow read: if isSales() || (isAuthenticated() && get(/databases/$(database)/documents/quotes/$(quoteId)).data.clientEmail == request.auth.token.email);
        allow create: if true;  // Created server-side with quote
        allow update, delete: if isSales();
      }
    }

    // orders: sales read/write; drivers read assigned orders
    match /orders/{orderId} {
      allow read: if isSales() || (isDriver() && resource.data.driverId == request.auth.uid);
      allow create, update, delete: if isSales();
    }

    // invoices: sales read/write; clients read own (by clientEmail)
    match /invoices/{invoiceId} {
      allow read: if isSales() || (isAuthenticated() && resource.data.clientEmail == request.auth.token.email);
      allow create, update, delete: if isSales();
    }

    // deliveries: sales read/write all; drivers read/write own
    match /deliveries/{deliveryId} {
      allow read, write: if isSales() || (isDriver() && (resource == null || resource.data.driverId == request.auth.uid));
    }

    // users: sales read/write; each user reads own doc
    match /users/{userId} {
      allow read: if isSales() || request.auth.uid == userId;
      allow create, update, delete: if isSales();
    }

    // counters: only server-side (admin) can write - no client access
    match /counters/{counterId} {
      allow read, write: if false;  // Server uses Admin SDK, bypasses rules
    }
  }
}
